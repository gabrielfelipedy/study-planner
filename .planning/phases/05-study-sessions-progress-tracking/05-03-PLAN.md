### Server Component (page.tsx)

```typescript
import { auth } from "@/lib/auth";
import { headers } from "next/headers";
import { redirect } from "next/navigation";
import { notFound } from "next/navigation";
import Link from "next/link";
import { ArrowLeft } from "lucide-react";
import { Button } from "@/components/ui/button";
import { getPlanById } from "@/lib/dal/queries/plans";
import { getTodaySchedule } from "@/lib/dal/queries/progress";
import { SessionChecklist } from "./session-checklist";
import { format } from "date-fns";

export default async function StudySessionPage({
  params,
}: {
  params: Promise<{ id: string }>;
}) {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session?.user?.id) redirect("/sign-in");

  const { id } = await params;
  const plan = await getPlanById(id, session.user.id);
  if (!plan) notFound();

  const today = format(new Date(), "yyyy-MM-dd");
  const todayTopics = await getTodaySchedule(session.user.id);
  const planTodayTopics = todayTopics.filter((t) => t.planId === id);

  return (
    <div className="mx-auto max-w-2xl px-4 py-8">
      {/* Header */}
      <div className="mb-6">
        <Link
          href={`/plans/${id}`}
          className="mb-4 inline-flex items-center gap-1 text-sm text-muted-foreground hover:text-foreground"
        >
          <ArrowLeft className="h-4 w-4" />
          Back to plan
        </Link>
        <h1 className="text-2xl font-semibold text-foreground">{plan.title}</h1>
        <p className="mt-1 text-sm text-muted-foreground">
          Study session — {today}
        </p>
      </div>

      {/* Progress summary */}
      <div className="mb-6">
        <div className="flex items-center justify-between text-sm">
          <span className="text-muted-foreground">
            {planTodayTopics.filter((t) => t.isCompleted).length} / {planTodayTopics.length} topics completed today
          </span>
          <span className="font-medium text-foreground">
            {planTodayTopics.length > 0
              ? `${Math.round((planTodayTopics.filter((t) => t.isCompleted).length / planTodayTopics.length) * 100)}%`
              : "—"}
          </span>
        </div>
        <div className="mt-1 h-2 w-full rounded-full bg-muted">
          <div
            className="h-2 rounded-full bg-primary transition-all"
            style={{
              width: planTodayTopics.length > 0
                ? `${(planTodayTopics.filter((t) => t.isCompleted).length / planTodayTopics.length) * 100}%`
                : "0%",
            }}
          />
        </div>
      </div>

      {/* Checklist */}
      {planTodayTopics.length > 0 ? (
        <SessionChecklist
          planId={id}
          topics={planTodayTopics}
        />
      ) : (
        <div className="rounded-lg border-2 border-dashed border-border p-12 text-center">
          <h3 className="text-lg font-medium text-foreground">No topics scheduled for today</h3>
          <p className="mt-2 text-sm text-muted-foreground">
            There are no study topics scheduled for today in this plan.
            Check your schedule or generate a timetable if you haven't yet.
          </p>
          <Link href={`/plans/${id}`}>
            <Button variant="default" className="mt-4">
              View schedule
            </Button>
          </Link>
        </div>
      )}
    </div>
  );
}
```

### Client Component (session-checklist.tsx)

Create alongside the page:
```typescript
"use client";

import { useState, useCallback } from "react";
import { useRouter } from "next/navigation";
import { CheckCircle2, Circle } from "lucide-react";
import type { TodayTopic } from "@/lib/dal/queries/progress";

type SessionChecklistProps = {
  planId: string;
  topics: TodayTopic[];
};

export function SessionChecklist({ planId, topics }: SessionChecklistProps) {
  const router = useRouter();
  const [markingIds, setMarkingIds] = useState<Set<string>>(new Set());
  const [completedIds, setCompletedIds] = useState<Set<string>>(
    new Set(topics.filter((t) => t.isCompleted).map((t) => t.topicId))
  );

  const handleMark = useCallback(async (topicId: string) => {
    if (markingIds.has(topicId)) return; // prevent double-click

    setMarkingIds((prev) => new Set(prev).add(topicId));

    try {
      const { markTopicStudiedAction } = await import("@/app/plans/[id]/actions");
      const result = await markTopicStudiedAction(planId, topicId);

      if (result.success) {
        setCompletedIds((prev) => new Set(prev).add(topicId));
        router.refresh();
      }
    } finally {
      setMarkingIds((prev) => {
        const next = new Set(prev);
        next.delete(topicId);
        return next;
      });
    }
  }, [planId, markingIds, router]);

  return (
    <div className="space-y-2">
      {topics.map((topic) => {
        const isDone = completedIds.has(topic.topicId);
        const isMarking = markingIds.has(topic.topicId);

        return (
          <div
            key={topic.slotId}
            className={`flex items-center gap-3 rounded-lg border bg-card p-4 transition-colors ${
              isDone ? "opacity-60" : ""
            }`}
          >
            {/* Status icon */}
            <button
              onClick={() => !isDone && handleMark(topic.topicId)}
              disabled={isDone || isMarking}
              className="shrink-0 focus:outline-none focus:ring-2 focus:ring-ring rounded-full"
              aria-label={isDone ? "Completed" : "Mark as studied"}
            >
              {isDone ? (
                <CheckCircle2 className="h-6 w-6 text-green-500" />
              ) : (
                <Circle className="h-6 w-6 text-muted-foreground hover:text-primary" />
              )}
            </button>

            {/* Topic info */}
            <div className="flex-1 min-w-0">
              <p className={`text-sm font-medium text-foreground truncate ${isDone ? "line-through" : ""}`}>
                {topic.topicTitle}
              </p>
              {topic.subjectName && (
                <div className="flex items-center gap-1.5 mt-0.5">
                  {topic.subjectColor && (
                    <div
                      className="h-2 w-2 rounded-full"
                      style={{ backgroundColor: topic.subjectColor }}
                    />
                  )}
                  <span className="text-xs text-muted-foreground">{topic.subjectName}</span>
                </div>
              )}
            </div>

            {/* Duration + action */}
            <div className="flex items-center gap-2 shrink-0">
              <span className="text-xs text-muted-foreground">{topic.estimatedMinutes}m</span>
              {!isDone && (
                <button
                  onClick={() => handleMark(topic.topicId)}
                  disabled={isMarking}
                  className="inline-flex items-center gap-1 rounded-md bg-primary px-3 py-1.5 text-xs font-medium text-primary-foreground hover:bg-primary/90 disabled:opacity-50"
                >
                  {isMarking ? "Marking..." : "Mark studied"}
                </button>
              )}
            </div>
          </div>
        );
      })}
    </div>
  );
}
```
